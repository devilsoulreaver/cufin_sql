select sli.empid, empname(sli.empid),sli.policyno, paybill.amount from paybill, ind_bills_master, sli where ind_bills_master.ind_bill_no = paybill.ind_bill_no and sli.empid= ind_bills_master.empid and ind_bills_master.ofdate='01/09/2021'  and paybill.indid='EXALLOWS'

select distinct ind_bills_master.empid, empname(ind_bills_master.empid), empoffice(ind_bills_master.empid), emp_master.doret from emp_master, ind_bills_master where emp_master.empid= ind_bills_master.empid and ofdate between '01/07/2019' and '01/03/2021' and ind_bills_master.btype='SAL' order by empid
