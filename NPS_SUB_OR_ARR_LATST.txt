select 
	ROW_NUMBER() OVER(ORDER BY (SELECT 1)) AS "SL NO",
	a.empid as "PF NO", b.prannumber as "PRAN NO",
	empname(a.empid) as "NAME", 
	c.amount as "EMPLOYEE'S CONTRIBUTION", 
	c.amount as "EMPLOYER'S CONTRIBUTION",
	c.amount*2 as "NPS TOTAL", 
	empbillsectid(a.empid) AS "BILL SECTION" 
from
	ind_bills_master a, emp_master b, paybill c, bills_det d
where 
	a.empid=b.empid and a.ind_bill_no=c.ind_bill_no
	and a.ind_bill_no = d.ind_billno
	and indid='NPS'
	--and indid='PEN'
	and billno between 2022001793 and 2022001865
	and btype IN ('SAL')
	order by 1, 3
